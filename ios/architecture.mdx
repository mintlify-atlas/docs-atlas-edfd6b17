## MVVM Pattern

The app implements the Model-View-ViewModel pattern for separation of concerns:

### Models

Models are lightweight, Codable structs that represent domain data:

```swift name/Models/Venue.swift
/// Represents a venue with full details
struct Venue: Codable, Identifiable, Hashable {
    let id: String
    let name: String
    let category: String
    let description: String
    let image: String
    let address: String
    let latitude: Double?
    let longitude: Double?
    let distance_km: Double?
}

/// Lightweight summary for list views
struct VenueListItem: Codable, Identifiable, Hashable {
    let id: String
    let name: String
    let category: String
    let image: String
    let interested_count: Int
    let distance_km: Double?
}
```

<Note>
**Design Decision:** Two separate models optimize for different use cases:
- `VenueListItem`: Minimal data for feed performance
- `Venue`: Full details loaded on demand
</Note>

### ViewModels

ViewModels manage state and business logic using Combine:

```swift name/ViewModels/VenueFeedViewModel.swift
@MainActor
class VenueFeedViewModel: ObservableObject {
    // Published properties for view binding
    @Published var venues: [VenueListItem] = []
    @Published var filters = VenueFilters()
    @Published var isLoading = false
    @Published var errorMessage: String?
    
    private let apiService: APIServiceProtocol
    private let appState: AppState
    
    init(apiService: APIServiceProtocol = APIService(), appState: AppState) {
        self.apiService = apiService
        self.appState = appState
    }
    
    /// Loads venues from the API with current filters
    func loadVenues() async {
        guard !isLoading else { return }
        
        isLoading = true
        errorMessage = nil
        
        do {
            let fetchedVenues = try await apiService.fetchVenues(
                userId: appState.currentUserId,
                filters: filters
            )
            self.venues = fetchedVenues
            self.isLoading = false
        } catch let error as APIError {
            self.errorMessage = error.errorDescription
            self.isLoading = false
        }
    }
}
```

<Tip>
**@MainActor:** All ViewModel methods are marked with `@MainActor` to ensure UI updates happen on the main thread, eliminating the need for manual `DispatchQueue.main.async` calls.
</Tip>

## App Entry Point

The app uses SwiftUI's modern App lifecycle:

```swift name/nameApp.swift
import SwiftUI

@main
struct nameApp: App {
    @StateObject private var themeManager = ThemeManager()
    @ObservedObject private var appState = AppState.shared
    private let notificationService = NotificationService.shared
    
    var body: some Scene {
        WindowGroup {
            ContentView()
                .environmentObject(themeManager)
                .preferredColorScheme(themeManager.colorScheme)
                .onOpenURL { url in
                    // Handle deep links (luna://venues/{id})
                    appState.handleDeepLink(url)
                }
                .task {
                    // Check for expired action items on app launch
                    await ActionItemExpirationManager.checkExpiredItems()
                    
                    // Request notification permission
                    _ = await notificationService.requestPermission()
                }
        }
    }
}
```

### Key Features

- **Theme Management**: `ThemeManager` handles dark mode with persistence
- **Global State**: `AppState` singleton manages user data and interests
- **Deep Linking**: Supports `luna://` URL scheme for navigation
- **Background Tasks**: Checks expired action items and requests permissions on launch

## Navigation Structure

The app uses a custom tab-based navigation system:

```swift name/ContentView.swift
struct ContentView: View {
    @ObservedObject private var appState = AppState.shared
    
    var body: some View {
        ZStack {
            // Current view based on selectedTab
            Group {
                switch appState.selectedTab {
                case 0: VenueFeedView()
                case 1: RecommendedFeedView()
                case 2: ActionTabView()
                case 3: ProfileView()
                default: VenueFeedView()
                }
            }
            
            // Custom floating tab bar
            if !appState.isTabBarHidden {
                VStack {
                    Spacer()
                    CustomTabBarView(
                        selectedTab: $appState.selectedTab,
                        socialBadgeCount: appState.newSocialActivityCount,
                        profileBadgeCount: appState.actionItemCount
                    )
                }
                .zIndex(100)
            }
            
            // Global Action Item Toast
            ActionItemToast(
                actionItem: $appState.pendingActionItem,
                isShowing: $appState.showActionItemToast
            )
            .zIndex(999)
        }
    }
}
```

## Dependency Injection

The app uses protocol-based dependency injection for testability:

```swift
protocol APIServiceProtocol {
    func fetchVenues(userId: String?, filters: VenueFilters?) async throws -> [VenueListItem]
    func fetchVenueDetail(venueId: String) async throws -> VenueDetailResponse
    func expressInterest(userId: String, venueId: String) async throws -> InterestResponse
}

class VenueFeedViewModel: ObservableObject {
    private let apiService: APIServiceProtocol
    
    // Inject real or mock service
    init(apiService: APIServiceProtocol = APIService()) {
        self.apiService = apiService
    }
}
```

<Accordion title="Why Protocol-Based Design?">
- **Testability**: Easily swap real services with mocks in unit tests
- **Flexibility**: Change implementations without modifying ViewModels
- **Decoupling**: ViewModels depend on abstractions, not concrete implementations
</Accordion>

## Thread Safety

The app ensures thread-safe UI updates using modern Swift concurrency:

```swift
@MainActor
class AppState: ObservableObject {
    @Published var currentUserId: String = ""
    @Published var interestedVenueIds: Set<String> = []
    
    func toggleInterest(venueId: String) async throws -> InterestResponse {
        // Optimistic update on main thread
        if interestedVenueIds.contains(venueId) {
            interestedVenueIds.remove(venueId)
        } else {
            interestedVenueIds.insert(venueId)
        }
        
        // Network call happens off main thread
        let response = try await apiService.expressInterest(
            userId: currentUserId,
            venueId: venueId
        )
        
        // Automatically back on main thread due to @MainActor
        return response
    }
}
```

<Warning>
**Legacy Code:** Avoid manual thread management with `DispatchQueue.main.async`. Use `@MainActor` and `async/await` instead for cleaner, safer code.
</Warning>

## Best Practices

<AccordionGroup>
  <Accordion title="State Management">
    - Use `@State` for local view state
    - Use `@StateObject` for view-owned ViewModels
    - Use `@ObservedObject` for shared ViewModels (like AppState)
    - Use `@EnvironmentObject` for app-wide dependencies (like ThemeManager)
  </Accordion>
  
  <Accordion title="Async/Await">
    - Prefer `async/await` over completion handlers
    - Mark ViewModel methods with `@MainActor` for UI safety
    - Use `Task {}` for calling async code from sync context
    - Handle errors with proper `do-catch` blocks
  </Accordion>
  
  <Accordion title="Performance">
    - Use `VenueListItem` for lists, `Venue` for details
    - Implement caching layers (memory + disk)
    - Use lazy loading for images
    - Avoid unnecessary view re-renders with proper state management
  </Accordion>
</AccordionGroup>
