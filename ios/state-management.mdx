The Luna iOS app uses a centralized state management approach with the `AppState` singleton and Combine framework for reactive updates.

## AppState Singleton

The `AppState` class is the single source of truth for global application state:

```swift name/ViewModels/AppState.swift
@MainActor
class AppState: ObservableObject {
    // Singleton instance
    nonisolated(unsafe) static let shared = AppState()
    
    // Published properties for reactive updates
    @Published var currentUserId: String = ""
    @Published var currentUserName: String = ""
    @Published var currentUserAvatar: String = ""
    @Published var interestedVenueIds: Set<String> = []
    @Published var actionItemCount: Int = 0
    @Published var selectedTab: Int = 0
    @Published var isTabBarHidden: Bool = false
    @Published var socialFeedActivities: [InterestActivity] = []
    @Published var highlightedVenues: [HighlightedVenue] = []
    @Published var newSocialActivityCount: Int = 0
    
    // Action item toast state
    @Published var pendingActionItem: ActionItem?
    @Published var showActionItemToast = false
    
    private let apiService: APIServiceProtocol
    private let authService: AuthenticationServiceProtocol
    
    nonisolated private init(
        apiService: APIServiceProtocol = APIService(),
        authService: AuthenticationServiceProtocol = MockAuthenticationService.shared
    ) {
        self.apiService = apiService
        self.authService = authService
        
        Task { @MainActor in
            self.currentUserId = authService.getCurrentUserId()
            self.updateCurrentUserInfo()
            await loadUserInterests()
        }
    }
}
```

<Note>
**@MainActor:** The entire `AppState` class is isolated to the main actor, ensuring all property updates happen on the main thread for safe UI updates.
</Note>

## Interest Management

The `toggleInterest` method demonstrates optimistic updates and error handling:

```swift name/ViewModels/AppState.swift
@discardableResult
func toggleInterest(venueId: String, venueInfo: ActivityVenue? = nil) async throws -> InterestResponse {
    let wasInterested = interestedVenueIds.contains(venueId)
    
    // Optimistic update - immediate UI feedback
    if wasInterested {
        interestedVenueIds.remove(venueId)
    } else {
        interestedVenueIds.insert(venueId)
    }
    
    let socialAction: InterestAction = wasInterested ? .notInterested : .interested
    
    do {
        // Call API to persist change
        let response = try await apiService.expressInterest(
            userId: currentUserId,
            venueId: venueId
        )
        
        // Broadcast to social feed
        await broadcastInterestToFriends(
            venueId: venueId,
            action: socialAction,
            venue: venueInfo
        )
        
        // Handle action item response
        if let actionItemData = response.action_item,
           actionItemData.action_item_created,
           let itemId = actionItemData.action_item_id {
            let actionItem = ActionItem(
                id: itemId,
                venue_id: venueId,
                interested_user_ids: actionItemData.interested_user_ids ?? [],
                action_type: "book_venue",
                action_code: actionItemData.action_code ?? "",
                description: actionItemData.description ?? "",
                threshold_met: true,
                status: "pending",
                created_at: ISO8601DateFormatter().string(from: Date()),
                venue: nil
            )
            
            showActionItemNotification(actionItem)
        }
        
        return response
    } catch {
        // Revert optimistic update on error
        if wasInterested {
            interestedVenueIds.insert(venueId)
        } else {
            interestedVenueIds.remove(venueId)
        }
        throw error
    }
}
```

<Accordion title="Optimistic Updates Explained">
**Optimistic updates** improve perceived performance by updating the UI immediately before waiting for the API response:

1. **Immediate Update**: Modify local state instantly (add/remove from `interestedVenueIds`)
2. **API Call**: Send request to backend
3. **Success**: Keep the optimistic update
4. **Failure**: Revert the change to maintain data consistency

This creates a snappy, responsive UI while maintaining data integrity.
</Accordion>

## User Switching

The app supports switching between multiple users:

```swift name/ViewModels/AppState.swift
func switchUser(to userId: String) async {
    // First, logout to clear all state
    logout()
    
    // Set new user
    authService.switchUser(to: userId)
    currentUserId = authService.getCurrentUserId()
    updateCurrentUserInfo()
    
    // Reload data for new user

    // Check for expired items

func logout() {
    // Clear user identity
    currentUserId = ""
    currentUserName = ""
    currentUserAvatar = ""
    
    // Clear interests
```

## Social Feed Broadcasting

When users express interest, the app broadcasts to friends' social feeds:

```swift name/ViewModels/AppState.swift
func broadcastInterestToFriends(
    venueId: String,
    action: InterestAction,
    venue: ActivityVenue?
) async {
    if action == .interested, let venue = venue {
        let displayName = currentUserName.isEmpty ? "Unknown User" : currentUserName
        
        let newActivity = InterestActivity(
            id: UUID().uuidString,
            user: ActivityUser(
                id: currentUserId,
                name: displayName,
                avatar: currentUserAvatar
            ),
            venue: venue,
            action: action,
            timestamp: Date(),
            isActive: true
        )
        
        // Add to local feed
        socialFeedActivities.insert(newActivity, at: 0)
        
        // Check if venue should be highlighted
        updateHighlightedVenues(for: venueId, action: action)
    } else if action == .notInterested {
        // Remove activity when user removes interest
        socialFeedActivities.removeAll { 
            $0.venue.id == venueId && $0.user.id == currentUserId 
        }
        
        updateHighlightedVenues(for: venueId, action: action)
    }
}
```

## Deep Link Handling

The app supports deep links with the `luna://` URL scheme:

```swift name/ViewModels/AppState.swift
func handleDeepLink(_ url: URL) {
    guard url.scheme == "luna" else { return }
    
    let pathComponents = url.pathComponents.filter { $0 != "/" }
    
    // Parse luna://venues/{venue_id}
    if pathComponents.count == 2,
       pathComponents[0] == "venues" {
        deepLinkVenueId = pathComponents[1]
        print("🔗 Deep link: Opening venue \(pathComponents[1])")
    }
}
```

**Integration in App:**

```swift name/nameApp.swift
@main
struct nameApp: App {
    @ObservedObject private var appState = AppState.shared
    
    var body: some Scene {
        WindowGroup {
            ContentView()
                .onOpenURL { url in
                    appState.handleDeepLink(url)
                }
        }
    }
}
```

**Handling in Views:**

```swift
struct VenueFeedView: View {
    @ObservedObject private var appState = AppState.shared
    @State private var showVenueDetail = false
    
    var body: some View {
        NavigationView {
            // Feed content
        }
        .onChange(of: appState.deepLinkVenueId) { venueId in
            if let venueId = venueId {
                // Navigate to venue detail
                showVenueDetail = true
            }
        }
        .sheet(isPresented: $showVenueDetail) {
            if let venueId = appState.deepLinkVenueId {
                VenueDetailView(venueId: venueId)
            }
        }
    }
}
```

## ViewModel Pattern

ViewModels manage screen-specific state using `@Published` properties:

```swift name/ViewModels/VenueFeedViewModel.swift
@MainActor
class VenueFeedViewModel: ObservableObject {
    @Published var venues: [VenueListItem] = []
    @Published var filters = VenueFilters()
    @Published var showFilterSheet = false
    @Published var isLoading = false
    @Published var errorMessage: String?
    @Published var lastUpdated: Date?
    
    private let apiService: APIServiceProtocol
    private let appState: AppState
    
    init(apiService: APIServiceProtocol = APIService(), appState: AppState) {
        self.apiService = apiService
        self.appState = appState
    }
    
    func loadVenues() async {
        guard !isLoading else { return }
        
        isLoading = true
        errorMessage = nil
        
        do {
            let fetchedVenues = try await apiService.fetchVenues(
                userId: appState.currentUserId,
                filters: filters
            )
            self.venues = fetchedVenues
            self.lastUpdated = Date()
            self.isLoading = false
        } catch let error as APIError {
            self.errorMessage = error.errorDescription
            self.isLoading = false
        }
    }
}
```

**View Integration:**

```swift
struct VenueFeedView: View {
    @StateObject private var viewModel = VenueFeedViewModel(appState: AppState.shared)
    
    var body: some View {
        NavigationView {
            Group {
                if viewModel.isLoading {
                    ProgressView()
                } else if let error = viewModel.errorMessage {
                    ErrorView(message: error)

                            ForEach(viewModel.venues) { venue in
                                VenueCardView(venue: venue)
                            }
                        }
                    }
                }
            }
        }
        .task {
            await viewModel.loadVenues()
        }
    }
}
```

```swift

class ProfileViewModel: ObservableObject {
    @Published var user: User?
```

## State Property Wrappers

<Tabs>
  <Tab title="@State">
    Local view state that persists across view updates:
    ```swift
    @State private var isExpanded = false
    @State private var selectedIndex = 0
    ```
    **Use for:** Animation state, temporary UI state, local toggles
  </Tab>
  
  <Tab title="@StateObject">
    Creates and owns an ObservableObject:
    ```swift
    @StateObject private var viewModel = VenueFeedViewModel()
    ```
    **Use for:** View-owned ViewModels that persist across view updates
  </Tab>
  
  <Tab title="@ObservedObject">
    Observes an existing ObservableObject:
    ```swift
    @ObservedObject private var appState = AppState.shared
    ```
    **Use for:** Shared state objects like AppState (not owned by view)
  </Tab>
  
  <Tab title="@EnvironmentObject">
    Dependency injected from environment:
    ```swift
    @EnvironmentObject var themeManager: ThemeManager
    ```
    **Use for:** App-wide dependencies passed down the view hierarchy
  </Tab>
</Tabs>

## Best Practices

<AccordionGroup>
  <Accordion title="Use AppState for Global State">
    Store user identity, interests, and navigation state in `AppState`:
    ```swift
    // ✅ Good
    @ObservedObject private var appState = AppState.shared
    let isInterested = appState.isInterested(in: venueId)
    
    // ❌ Bad
    @State private var isInterested = false // Doesn't sync across views
    ```
  </Accordion>
  
  <Accordion title="Use ViewModels for Screen State">
    Screen-specific state belongs in ViewModels:
    ```swift
    // ✅ Good
    @StateObject private var viewModel = VenueFeedViewModel()
    
    // ❌ Bad - storing screen state in AppState
    appState.venuesList = [...]
    ```
  </Accordion>
  
  <Accordion title="Optimistic Updates for UX">
    Update UI immediately, then sync with API:
    ```swift
    // Update local state
    interestedVenueIds.insert(venueId)
    
    // Then sync with API
    do {
        try await apiService.expressInterest(...)
    } catch {
        // Revert on error
        interestedVenueIds.remove(venueId)
    }
    ```
  </Accordion>
  
  <Accordion title="@MainActor for UI Safety">
    Mark classes with `@MainActor` to ensure thread-safe UI updates:
    ```swift
    @MainActor
    class AppState: ObservableObject {
        // All updates happen on main thread
    }
    ```
  </Accordion>
</AccordionGroup>
